What would be the output for the following program:

package main

import "fmt"

func main() {
        var a, b = 100, 5
        switch {
        case a/b == 10:
                fmt.Println("10")
        case a/b == 20:
                fmt.Println("20")
        case a/b == 10:
                fmt.Println("30")
        default:
                fmt.Println("default")
        }

}

==> 20

What would be the output for the following program:


Note: When using switch with conditionals - we don't specify expression after switch.

package main

import "fmt"

func main() {
        var i, j = 10, 50

        switch {
        case i+j == 60:
                fmt.Println("equal to 60")
        case i+j <= 60:
                fmt.Println("less than or equal to 60")
                fallthrough
        default:
                fmt.Println("greater than 60")
        }

}
==> equal to 60
